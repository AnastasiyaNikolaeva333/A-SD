{
Дан текстовый файл. Произвести его шифрацию по следующим правилам:
     1) символы с кодами 1-31,243-255 не менять, т.к. они могут играть роль служебных;
     2) шифровать текст блоками по 32 байта;
     3) коды символов 32-242 зашифровать по формулам
            a[n]=((243-n+m) mod 211)+32
            m=(3*k-1) mod 99               , где
        n-исходный код символа;
        a[n]-зашифрованный код;
        k-порядковый номер блока.
     Как произвести дешифрацию текста (7)?
}

PROGRAM Encryption(INPUT, OUTPUT);
TYPE
  ByteMassiv = ARRAY[1..32] OF BYTE;
  FileType = FILE OF BYTE;
VAR
  InFile, OutFile, Mode: STRING;
  InputFile, OutputFile: FileType;

PROCEDURE ShowError(Error: STRING);
BEGIN
  WRITELN('Ошибка: ', Error);
  WRITELN('Нажмите Enter для выхода из программы');
  READLN;
  HALT(1);
END;

PROCEDURE ErrorFile(SendFilee: STRING);
VAR
  ErrorCode: INTEGER;
BEGIN
  ErrorCode := IORESULT;
  IF ErrorCode > 0 
  THEN
    ShowError('Файл "' + SendFilee + '" не найден')
END;

PROCEDURE GetFiles(VAR InputFile, OutputFile: FileType);
BEGIN
  WRITELN('Введите входной файл: ');
  READLN(InFile);
  IF InFile = '' 
  THEN
    ShowError('Не указано имя входного файла')
  ELSE
    BEGIN
      ASSIGN(InputFile, InFile); 
      {$I-}
      RESET(InputFile);
      ErrorFile(InFile)
      {$I+}
    END;
    
  WRITELN('Введите выходной файл: ');
  READLN(OutFile);  
  IF OutFile = '' 
  THEN
    ShowError('Не указано имя выходного файла')
  ELSE
    BEGIN
      IF InFile = OutFile 
      THEN
        ShowError('Входной и выходной файлы не должны совпадать');
      ASSIGN(OutputFile, OutFile);
      {$I-}
      REWRITE(OutputFile);
      ErrorFile(OutFile)
      {$I+}
    END
END;

PROCEDURE EncryptionBlock(VAR Block: ByteMassiv; CountBlock, CoutByte: INTEGER);
VAR
  I, m: INTEGER;
BEGIN
  FOR I := 1 TO CoutByte 
  DO
    BEGIN
      IF (Block[I] >= 32) AND (Block[I] <= 242)
      THEN
        BEGIN
          m := (3 * CountBlock - 1) MOD 99; 
          Block[I] := ((243 - Block[I] + m) MOD 211) + 32
        END    
    END 
END;

PROCEDURE DecryptionBlock(VAR Block: ByteMassiv; CountBlock, CoutByte: INTEGER);
VAR
  I, m: INTEGER;
BEGIN
  FOR I := 1 TO CoutByte 
  DO
    BEGIN
      IF (Block[I] >= 32) AND (Block[I] <= 242)
      THEN
        BEGIN
          m := (3 * CountBlock - 1) MOD 99;
          Block[I] := (275 + m - Block[I]) MOD 211;
          IF Block[I] < 32
          THEN
            Block[I] := Block[I] + 211  
        END  
    END 
END;

PROCEDURE PrintBlock(VAR OutputFile: FileType; VAR Block: ByteMassiv; CoutByte: INTEGER);
VAR
  I: INTEGER;
BEGIN
  FOR I := 1 TO CoutByte 
  DO
    WRITE(OutputFile, Block[I])
END;

PROCEDURE WorkWithFile(VAR InputFile, OutputFile: FileType; Flag: BOOLEAN);
VAR
  Block: ByteMassiv;
  I, CoutByte, CountBlock: INTEGER;
  ByteValue: BYTE;
BEGIN
  CountBlock := 0;
  WHILE NOT EOF(InputFile) 
  DO
    BEGIN
      CoutByte := 0;
      CountBlock := CountBlock + 1;
      FOR I := 1 TO 32 
      DO
        BEGIN
          IF NOT EOF(InputFile) 
          THEN
            BEGIN
              READ(InputFile, ByteValue);
              Block[I] := ByteValue;
              CoutByte := CoutByte + 1
            END
          ELSE
            BREAK
        END;
    
      IF Flag 
      THEN
        EncryptionBlock(Block, CountBlock, CoutByte)
      ELSE
        DecryptionBlock(Block, CountBlock, CoutByte);  
    
      PrintBlock(OutputFile, Block, CoutByte)
    END
END;

BEGIN
  GetFiles(InputFile, OutputFile);

  WRITELN('Выберите режим работы:');
  WRITELN('1 - Шифрование');
  WRITELN('2 - Дешифрование');
  WRITE('Введите 1 или 2: ');
  READLN(Mode);
  
  IF (Mode <> '1') AND (Mode <> '2') 
  THEN
    ShowError('Неверный выбор режима');
    
  IF Mode = '1' 
  THEN
    BEGIN
      WRITELN('Режим: ШИФРОВАНИЕ');
      WorkWithFile(InputFile, OutputFile, TRUE)
    END
  ELSE
    BEGIN
      WRITELN('Режим: ДЕШИФРОВАНИЕ');
      WorkWithFile(InputFile, OutputFile, FALSE)
    END;

  CLOSE(InputFile);
  CLOSE(OutputFile);
  
  WRITELN('Обработка завершена успешно!');
  WRITELN('Нажмите Enter для выхода...');
  READLN
END.
